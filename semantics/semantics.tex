\subsubsection*{Grammar}

\begin{mathpar}

  \grammar
  { \gramrule{\t}{Base type} \\
    \gramitem{\bool} \\
    \gramitem{\uint{n}} \\
    \gramitem{\int{n}} \\
    \gramitem{\arr{\t,n}} }

  \grammar
  { \gramrule{\l}{Label} \\
    \gramitem{\public} \\
    %\gramitem{\unknown} \\
    \gramitem{\secret} }

  \grammar
  { \gramrule{\k}{Mutability} \\
    \gramitem{\const} \\
    \gramitem{\mut} }

  \grammar
  { \gramrule{v}{Function argument} \\
    \gramitem[by value]{e} \\
    \gramitem[by reference]{\mut\ x} }

  \grammar
  { \gramrule{e}{Expression} \\
    \gramitem{\ktrue} \\
    \gramitem{\kfalse} \\
    \gramitem[integer literal]{k} \\
    \gramitem[variable]{x} \\
    \gramitem[array get]{a[e]} \\
    \gramitem[unary op]{\ominus e} \\
    \gramitem[binary op]{e_1 \oplus e_2} \\
    \gramitem[function call]{f(v_1,\dots,v_n)} }

  \grammar
  { \gramrule{init}{ArrInit} \\
    \gramitem[memset zero]{\vc{\t,n}\textsc{zeros}} \\
    \gramitem[memset]{\vc{\t,n}\textsc{fill}(e)} \\
    \gramitem[array comprehension]{\vc{\t,n}x \Rightarrow e} \\
    \gramitem[literal]{\vc{\t,n}\{e_1,\dots,e_n\}} \\
    \gramitem[memcpy]{\textsc{copy}(a)} }

  \grammar
  { \gramrule{s}{Statement} \\
    \gramitem[sequence]{s_1 ; s_2} \\
    \gramitem[variable declaration]{\vc{\t,\l,\k} x := e} \\
    \gramitem[array declaration]{\vc{\arr{\t,n},\l,\k} a := init} \\
    \gramitem[array view declaration]{\vc{\arr{\t,n},\l,\k} a := \arrview{a',e}} \\
    \gramitem[variable assignment]{x := e} \\
    \gramitem[array assignment]{a[e_1] = e_2} \\
    \gramitem[conditional]{\kif e\ \{ s_1 \} \kelse \{ s_2 \}} \\
    \gramitem[loop]{\kfor \vc{\t} x \kfrom e_1 \kto e_2\ \{ s \}} \\
    \gramitem[return]{\kret e} }

\end{mathpar}

\subsubsection*{Type Lattice}
\begin{mathpar}

  \infer
  { n_1 < n_2 }
  { \uint{n_1} <_\t \uint{n_2} }

  \infer
  { n_1 < n_2 }
  { \int{n_1} <_\t \int{n_2} }

  \infer { }
  { \uint{n} <_\t \int{2n} }

  \infer { }
  { \public <_\l \secret }

  \infer { }
  { \const <_\k \mut }

  \infer
  { \t_1 <_\t \t_2 \\
    \G \has e : \vc{\t_1, \l} }
  { \G \has e : \vc{\t_2, \l} }

  \infer { }
  { \l \cup \l = \l }

  \infer { }
  { \l \cup \secret = \secret }

\end{mathpar}

\subsubsection*{Parameter Passing}
\begin{mathpar}

  \infer
  { \G \has e : \vc{\t,\l_1} \\
    \l_1 \leq_\l \l_2 }
  { \vc{\t,\l_2,\const} \gets e }

  \infer
  { \m (x) = \vc{\t,\l,\mut} }
  { \vc{\t,\l,\mut} \gets \mut\ x }

\end{mathpar}

\subsubsection*{Expressions}
\begin{mathpar}

  \infer[Var]
  { \m (x) = \vc{\t,\l,\k} }
  { \G \has x : \vc{\t,\l} }

  \infer[Unop]
  { \G \has e : \vc{\t_1,\l_1} \\
    \ominus : \vc{\t_1,\l_1} \to \vc{\t_2,\l_2} }
  { \G \has \ominus e : \vc{\t_2,\l_2} }

  \infer[Binop]
  { \G \has e_1 : \vc{\t_1,\l_1} \\
    \G \has e_2 : \vc{\t_2,\l_2} \\
    \oplus : \vc{\t_1,\l_1} \to \vc{\t_2,\l_2} \to \vc{\t_3,\l_3} }
  { \G \has e_1 \oplus e_2 : \vc{\t_3, \l_3 } }

  % Instead of dynamic looping or masking,
  % just fall back to static analysis;
  % if we cannot determine statically that it's ok
  % then the program fails typechecking
  \infer[ArrGet]
  { \m (a) = \vc{\arr{\t,n},\l,\k} \\
    \G \has e : \vc{\uint{max},\public} }
  { \G \has a[e] : \vc{\t,\l} }

  \infer[FnCall]
  { \F (f) = fdec (p_1, \dots, p_n) : \vc{\t,\l} \\
    p_1 \gets v_1 \\ \cdots \\ p_n \gets v_n }
  { \G \has f(v_1,\dots,v_n) : \vc{\t,\l} }

  \infer[True] { } { \G \has \ktrue : \vc{bool,\public} } \\
  \infer[False] { } { \G \has \kfalse : \vc{bool,\public} }

  \infer[PosNumber]
  { k >= 0 \\
    n = \ceil{\log_2 k} }
  { \G \has k : \vc{\uint{n},\public} }

  \infer[NegNumber]
  { k < 0 \\
    n = \ceil{\log_2 \abs{k}}  + 1 }
  { \G \has k : \vc{\int{n},\public} }

\end{mathpar}

\subsubsection*{Array Initializers}
\begin{mathpar}

  \infer[ZeroInit]
  { \t = \uint{s} \lor \t = \int{s} }
  { \G \has \vc{\t,n}\textsc{zeros} : \vc{\arr{\t,n},\public} }

  \infer[FillInit]
  { \G \has e : \vc{\t,\l} }
  { \G \has \vc{\t,n}\textsc{fill}(e) : \vc{\arr{\t,n},\l} }

  \infer[CompInit]
  { \G \has e : \vc{\t,\l} }
  { \G \has \vc{\t,n}x \Rightarrow e : \vc{\arr{\t,n},\l} }

  \infer[LitInit]
  { \G \has e_i : \vc{\t,\l_i} \\
    \textstyle\bigcup_i \l_i = \l }
  { \G \has \vc{\t,n}\{e_1,\dots,e_n\} : \vc{\arr{\t,n},\l} }

  \infer[CopyInit]
  { \m (a) = \vc{\arr{\t,n},\l,\k} }
  { \G \has \textsc{copy}(a) : \vc{\arr{\t,n},\l} }

\end{mathpar}

\subsubsection*{Statements}
\begin{mathpar}

  \newcommand\stmtitle{}
  \newcommand\stt[1]{\renewcommand\stmtitle{$#1 \stmto$}}
  \newcommand\xinfer[3]{\stt{#1}\infer[\stmtitle]{#2}{#3}}

  \infer[Seq]
  { \S\vc{\l_s} \has s_1 : \l'_s \\
    \S\vc{\l'_s} \has s_2 : \l''_s }
  { \S\vc{\l_s} \has s_1 ; s_2 : \l'_s \cup \l''_s }

  \infer[VarDec]
  { \G \has e : \vc{\t,\l_1} \\
    \t \neq \arr{\t',n} \\
    \l_1 \leq_\l \l_2 }
  { \S\vc{\l_s} \has \vc{\t,\l_2,\k} x := e : \public \\\\
    \mu (x) = \vc{\t,\l_2,\k} \text{\tiny\ (scoping?)}}

  \infer[ArrDec]
  { \G \has init : \vc{\arr{\t,n},\l_1} \\
    \l_1 \leq_\l \l_2 }
  { \S\vc{\l_s} \has \vc{\arr{\t,n},\l_2,\k} a := init : \public \\\\
    \mu (a) = \vc{\arr{\t,n},\l_2,\k} \text{\tiny\ (scoping?)}}

  \infer[ArrView]
  { \m (a) = \vc{\arr{\t,n},\l,\k} \\
    \G \has e : \vc{\uint{max},\public} \\
    n' \leq n \\
    \l \leq_\l \l' \\
    \k' \leq_\k \k }
  { \S\vc{\l_s} \has \vc{\arr{\t,n'},\l',\k'} a' := \arrview{a,e} : \public \\\\
    \mu (a') = \vc{\arr{\t,n'},\l',\k'} \text{\tiny\ (scoping?)}}

  \infer[VarAssign]
  { \mu (x) = \vc{\t,\l_1,\mut} \\
    \t \neq \arr{\t',n} \\
    \G \has e : \vc{\t,\l_2} \\
    \l_2 \leq_\l \l_1 }
  { \S\vc{\l_s} \has x := e : \public }

  \infer[ArrAssign]
  { \mu (a) = \vc{\arr{\t,n},\l_1,\mut} \\
    \G \has e_1 : \vc{\uint{max},\public} \\
    \G \has e_2 : \vc{\t,\l_2} \\
    \l_2 \leq_\l \l_1 }
  { \S\vc{\l_s} \has a[e_1] := e_2 : \public }

  \infer[If]
  { \G \has e : \vc{\bool,\l} \\
    \S\vc{\l \cup \l_s} \has s_1 : \l'_s \\
    \S\vc{\l \cup \l_s} \has s_2 : \l''_s }
  { \S\vc{\l_s} \has \kif e\ \{ s_1 \} \kelse \{ s_2 \} : \l'_s \cup \l''_s }

  \infer[For]
  { \G \has e_1 : \vc{\t,\public} \\
    \G \has e_2 : \vc{\t,\public} \\
    \t = \uint{s} \lor \t = \int{s} \\
    \S\vc{\l_s} \has s : \l'_s }
  { \S\vc{\l_s} \has \kfor \vc{\t} x \kfrom e_1 \kto e_2\ \{ s \} : \l'_s \\\\
    \mu (x) = \vc{\t,\public,\const} \text{\tiny\ (scoping?)}}

  \infer[Ret]
  { \G \has e : \vc{\t,\l_1} \\
    \F (f) = fdec : \vc{\t,\l_2} \\
    \l_1 \leq_\l \l_2 }
  { \S\vc{\l_s} \has \kret e : \l_s }

\end{mathpar}
